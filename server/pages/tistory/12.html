버튼을 누르면 Hello world를 출력하는 간단한 예제를 만들어봤다.

<?xml version="1.0" encoding="utf-8"?>
<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml" layout="absolute">
    <mx:Script>
        <![CDATA[
            import mx.controls.Alert ;
            private function clickBtn (mouseEvent :MouseEvent) :void
            {
                Alert.show("Hello, world!") ;
            }
        ]]>
    </mx:Script>
    <mx:Panel width='400' height='300' layout='vertical' horizontalAlign="center" verticalAlign="middle" horizontalCenter="0" verticalCenter="1" id='myPanel'>
       
        <mx:Button label='Button' click="clickBtn(event)" />
    </mx:Panel>   
</mx:Application>
